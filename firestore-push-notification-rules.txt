// Additional Firestore Security Rules for Push Notifications
// Add these to your firestore.rules file

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // FCM Tokens Collection
    // Users can only read/write their own FCM tokens
    match /fcmTokens/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // User Preferences (including notification preferences)
    // Users can only read/write their own preferences
    match /userPreferences/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Admin can read all FCM tokens for sending notifications
    match /fcmTokens/{userId} {
      allow read: if request.auth != null && 
                     exists(/databases/$(database)/documents/admins/$(request.auth.uid));
    }
    
    // For testing: Allow reads in development (remove in production)
    // match /fcmTokens/{userId} {
    //   allow read: if true;
    // }
  }
}
